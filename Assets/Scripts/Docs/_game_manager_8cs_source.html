301 private double HaversineDistance(double lat1, double lon1, double lat2, double lon2)
302 {
303 double R = 6371000; // Earth radius in meters
304 double dLat = Mathf.Deg2Rad * (float)(lat2 - lat1);
305 double dLon = Mathf.Deg2Rad * (float)(lon2 - lon1);
306
307 double a =
308 Mathf.Sin((float)dLat / 2) * Mathf.Sin((float)dLat / 2) +
309 Mathf.Cos(Mathf.Deg2Rad * (float)lat1) * Mathf.Cos(Mathf.Deg2Rad * (float)lat2) *
310 Mathf.Sin((float)dLon / 2) * Mathf.Sin((float)dLon / 2);
311
312 double c = 2 * Mathf.Atan2(Mathf.Sqrt((float)a), Mathf.Sqrt((float)(1 - a)));
313 double distance = R * c;
314 return distance;
315 }
